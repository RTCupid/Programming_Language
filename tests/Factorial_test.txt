{

    input (argument);

    answer = Factorial (argument);

    write_about {answer};

Factorial (fact)
{
    if_happens (fact < 2)
    {
        return 1;
    };
    if_happens (fact < 3)
    {
        return 2;
    };

    fact = fact * Factorial (fact - 1);

    return fact;
};

then_commit_seppuku
